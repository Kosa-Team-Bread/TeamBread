-- 1. tbl_admin: INT
CREATE TABLE tbl_admin (
    admin_id   INT PRIMARY KEY,
    admin_pw   VARCHAR2(100) NOT NULL,
    admin_name VARCHAR2(20)  NOT NULL,
    grade      NUMBER(1)     NOT NULL,
    regDate    DATE          NOT NULL,
    modDate    DATE          NOT NULL,
    email      VARCHAR2(100) NOT NULL
);

-- 2. tbl_category
CREATE TABLE tbl_category (
    category_id       INT PRIMARY KEY,
    category_parentid INT,
    category_name     VARCHAR2(20) NOT NULL,
    category_level    INT          NOT NULL
);

-- 3. tbl_product
CREATE TABLE tbl_product (
    product_id     INT PRIMARY KEY,
    category_id    INT NOT NULL,
    name           VARCHAR2(100) NOT NULL,
    price          INT NOT NULL,
    cost           INT NOT NULL,
    product_regdate DATE NOT NULL,
    product_modDate DATE,
    FOREIGN KEY (category_id) REFERENCES tbl_category(category_id)
);

-- 4. tbl_image
CREATE TABLE tbl_image (
    image_id      INT PRIMARY KEY,
    product_id    INT NOT NULL,
    category_id   INT NOT NULL,
    image_name    VARCHAR2(20)  NOT NULL,
    image_location VARCHAR2(100) NOT NULL,
    image_regdate DATE          NOT NULL,
    image_moddate DATE          NOT NULL,
    FOREIGN KEY (product_id)  REFERENCES tbl_product(product_id),
    FOREIGN KEY (category_id) REFERENCES tbl_category(category_id)
);

-- 5. tbl_stock: admin_id VARCHAR2 → INT로 변경
CREATE TABLE tbl_stock (
    stock_id      INT PRIMARY KEY,
    admin_id      INT NOT NULL,
    product_id    INT NOT NULL,
    category_id   INT NOT NULL,
    quantity      INT NOT NULL,
    location      VARCHAR2(20) NOT NULL,
    stock_regDate DATE          NOT NULL,
    stock_modDate DATE          NOT NULL,
    FOREIGN KEY (admin_id)      REFERENCES tbl_admin(admin_id),
    FOREIGN KEY (product_id)    REFERENCES tbl_product(product_id),
    FOREIGN KEY (category_id)   REFERENCES tbl_category(category_id)
);

-- 6. tbl_storageRetrieval: admin_id VARCHAR2 → INT로 변경
CREATE TABLE tbl_storageRetrieval (
    input_id       INT PRIMARY KEY,
    stock_id       INT NOT NULL,
    admin_id       INT NOT NULL,
    product_id     INT NOT NULL,
    category_id    INT NOT NULL,
    input_type     NUMBER(1)     NOT NULL,
    inout_quantity INT           NOT NULL,
    inout_content  VARCHAR2(100) NOT NULL,
    inout_regdate  DATE          NOT NULL,
    FOREIGN KEY (stock_id)       REFERENCES tbl_stock(stock_id),
    FOREIGN KEY (admin_id)       REFERENCES tbl_admin(admin_id),
    FOREIGN KEY (product_id)     REFERENCES tbl_product(product_id),
    FOREIGN KEY (category_id)    REFERENCES tbl_category(category_id)
);

-- 7. tbl_coupon (변경 없음)
CREATE TABLE tbl_coupon (
    coupon_id   INT PRIMARY KEY,
    product_id  INT NOT NULL,
    category_id INT NOT NULL,
    coupon_name VARCHAR2(100) NOT NULL,
    percent     INT NOT NULL,
    starttime   DATE NOT NULL,
    deadline    DATE NOT NULL,
    FOREIGN KEY (product_id)  REFERENCES tbl_product(product_id),
    FOREIGN KEY (category_id) REFERENCES tbl_category(category_id)
);
