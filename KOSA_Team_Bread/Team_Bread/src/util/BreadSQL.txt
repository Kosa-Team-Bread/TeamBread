-- 1. tbl_admin: INT
CREATE TABLE tbl_admin (
    admin_id   INT PRIMARY KEY,
    admin_pw   VARCHAR2(100) NOT NULL,
    admin_name VARCHAR2(20)  NOT NULL,
    grade      NUMBER(1)     NOT NULL,
    regDate    DATE          NOT NULL,
    modDate    DATE          NOT NULL,
    email      VARCHAR2(100) NOT NULL
);

-- 2. tbl_category
CREATE TABLE tbl_category (
    category_id       INT PRIMARY KEY,
    category_parentid INT,
    category_name     VARCHAR2(20) NOT NULL,
    category_level    INT          NOT NULL
);

-- 3. tbl_product
CREATE TABLE tbl_product (
    product_id     INT PRIMARY KEY,
    category_id    INT NOT NULL,
    name           VARCHAR2(100) NOT NULL,
    price          INT NOT NULL,
    cost           INT NOT NULL,
    product_regdate DATE NOT NULL,
    product_modDate DATE,
    FOREIGN KEY (category_id) REFERENCES tbl_category(category_id)
);

-- 4. tbl_image
CREATE TABLE tbl_image (
    image_id      INT PRIMARY KEY,
    product_id    INT NOT NULL,
    category_id   INT NOT NULL,
    image_name    VARCHAR2(20)  NOT NULL,
    image_location VARCHAR2(100) NOT NULL,
    image_regdate DATE          NOT NULL,
    image_moddate DATE          NOT NULL,
    FOREIGN KEY (product_id)  REFERENCES tbl_product(product_id),
    FOREIGN KEY (category_id) REFERENCES tbl_category(category_id)
);

-- 5. tbl_stock: admin_id VARCHAR2 → INT로 변경
CREATE TABLE tbl_stock (
    stock_id      INT PRIMARY KEY,
    admin_id      INT NOT NULL,
    product_id    INT NOT NULL,
    category_id   INT NOT NULL,
    quantity      INT NOT NULL,
    stock_name    VARCHAR2(100) NOT NULL,
    location      VARCHAR2(200) NOT NULL,
    stock_regDate DATE          NOT NULL,
    stock_modDate DATE          NOT NULL,
    FOREIGN KEY (admin_id)      REFERENCES tbl_admin(admin_id),
    FOREIGN KEY (product_id)    REFERENCES tbl_product(product_id),
    FOREIGN KEY (category_id)   REFERENCES tbl_category(category_id)
);

-- 6. tbl_storageRetrieval: admin_id VARCHAR2 → INT로 변경
CREATE TABLE tbl_storageRetrieval (
    inout_id       INT PRIMARY KEY,
    stock_id       INT NOT NULL,
    admin_id       INT NOT NULL,
    product_id     INT NOT NULL,
    category_id    INT NOT NULL,
    inout_type     NUMBER(1)     NOT NULL,
    inout_quantity INT           NOT NULL,
    inout_content  VARCHAR2(100) NOT NULL,
    inout_regdate  DATE          NOT NULL,
    FOREIGN KEY (stock_id)       REFERENCES tbl_stock(stock_id),
    FOREIGN KEY (admin_id)       REFERENCES tbl_admin(admin_id),
    FOREIGN KEY (product_id)     REFERENCES tbl_product(product_id),
    FOREIGN KEY (category_id)    REFERENCES tbl_category(category_id)
);

-- 7. tbl_coupon (변경 없음)
CREATE TABLE tbl_coupon (
    coupon_id   INT PRIMARY KEY,
    product_id  INT NOT NULL,
    category_id INT NOT NULL,
    coupon_name VARCHAR2(100) NOT NULL,
    percent     INT NOT NULL,
    starttime   DATE NOT NULL,
    deadline    DATE NOT NULL,
    FOREIGN KEY (product_id)  REFERENCES tbl_product(product_id),
    FOREIGN KEY (category_id) REFERENCES tbl_category(category_id)
);

-- 상품 삽입 저장 프로시저 --
CREATE OR REPLACE PROCEDURE proc_add_product_with_stock_and_image (
    p_category_id    IN tbl_product.category_id%TYPE,
    p_name           IN tbl_product.name%TYPE,
    p_price          IN tbl_product.price%TYPE,
    p_cost           IN tbl_product.cost%TYPE,
    p_admin_id       IN tbl_stock.admin_id%TYPE,
    p_quantity       IN tbl_stock.quantity%TYPE,
    p_stock_name     IN tbl_stock.stock_name%TYPE,
    p_location       IN tbl_stock.location%TYPE,
    p_image_location IN tbl_image.image_location%TYPE  -- New parameter for image path
)
IS
BEGIN
    -- 1. Insert new product
    INSERT INTO tbl_product (
        product_id, category_id, name, price, cost, product_regdate
    ) VALUES (
        seq_product_id.NEXTVAL, p_category_id, p_name, p_price, p_cost, SYSDATE
    );
    
    -- 2. Insert initial stock for the new product
    INSERT INTO tbl_stock (
        stock_id, admin_id, product_id, category_id,
        quantity, stock_name, location, stock_regDate, stock_modDate
    ) VALUES (
        seq_stock_id.NEXTVAL, p_admin_id, seq_product_id.CURRVAL, p_category_id,
        p_quantity, p_stock_name, p_location, SYSDATE, SYSDATE
    );
    
    -- 3. Insert image record for the new product
    INSERT INTO tbl_image (
        image_id, product_id, category_id,
        image_name, image_location, image_regDate, image_modDate
    ) VALUES (
        seq_image_id.NEXTVAL, seq_product_id.CURRVAL, p_category_id,
        p_name, p_image_location, SYSDATE, SYSDATE
    );
    
    -- 4. Commit the transaction to save all changes
    COMMIT;
END;
/

-- 입출고 삽입시 재고 수량 업데이트 --
CREATE OR REPLACE PROCEDURE proc_insert_inout (
    p_stock_id        IN tbl_storageRetrieval.stock_id%TYPE,
    p_admin_id        IN tbl_storageRetrieval.admin_id%TYPE,
    p_product_id      IN tbl_storageRetrieval.product_id%TYPE,
    p_category_id     IN tbl_storageRetrieval.category_id%TYPE,
    p_inout_type      IN tbl_storageRetrieval.inout_type%TYPE, -- 1: 입고, 2: 출고
    p_inout_quantity  IN tbl_storageRetrieval.inout_quantity%TYPE,
    p_inout_content   IN tbl_storageRetrieval.inout_content%TYPE
)
IS
BEGIN
    -- 1. 입출고 기록
    INSERT INTO tbl_storageRetrieval (
        inout_id, stock_id, admin_id, product_id, category_id,
        inout_type, inout_quantity, inout_content, inout_regdate
    ) VALUES (
        seq_inout_id.NEXTVAL, p_stock_id, p_admin_id, p_product_id, p_category_id,
        p_inout_type, p_inout_quantity, p_inout_content, SYSDATE
    );

    -- 2. 재고 수량 업데이트
    IF p_inout_type = 1 THEN
        UPDATE tbl_stock
           SET quantity = quantity + p_inout_quantity,
               stock_modDate = SYSDATE
         WHERE stock_id = p_stock_id;
    ELSIF p_inout_type = 2 THEN
        UPDATE tbl_stock
           SET quantity = quantity - p_inout_quantity,
               stock_modDate = SYSDATE
         WHERE stock_id = p_stock_id;
    END IF;

    COMMIT;
END;
/

-- 상품 삭제시 이미지, 재고, 쿠폰, 상품삭제 --
CREATE OR REPLACE PROCEDURE proc_delete_product (
    p_product_id IN tbl_product.product_id%TYPE
)
IS
BEGIN
    -- 1. 이미지 삭제
    DELETE FROM tbl_image
     WHERE product_id = p_product_id;

    -- 2. 재고 삭제
    DELETE FROM tbl_stock
     WHERE product_id = p_product_id;

    -- 3. 쿠폰 삭제
    DELETE FROM tbl_coupon
     WHERE product_id = p_product_id;

    -- 4. 상품 삭제
    DELETE FROM tbl_product
     WHERE product_id = p_product_id;

    COMMIT;
END;
/
